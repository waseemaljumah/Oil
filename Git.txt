# 1️⃣ تحقق من حالة الملفات للتأكد من الملفات المعدلة
git status

# 2️⃣ أضف كل الملفات المعدلة إلى مرحلة التحضير
git add .

# 3️⃣ اعمل commit مع رسالة تصف التعديلات
git commit -m "تحديث النظام: تصحيح التاريخ، فرق الممشى، ترتيب المعدات، إضافة متابعة يومية"

# 4️⃣ ادفع التغييرات إلى المستودع على الفرع الرئيسي (main أو master حسب اسم الفرع لديك)
git push origin main
